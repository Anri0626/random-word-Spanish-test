let currentVerb; let wrongAnswers = []; function getRandomVerb() { return verbs[Math.floor(Math.random() * verbs.length)]; } function displayVerb() { currentVerb = getRandomVerb(); document.getElementById("verb").textContent = `次の動詞の種類は？： ${currentVerb.infinitive}`; document.getElementById("feedback").textContent = ""; document.getElementById("pattern").textContent = ""; document.getElementById("translation").textContent = ""; document.getElementById("example").textContent = ""; document.getElementById("conjugation-table").innerHTML = "選択後に表示されます"; } function checkAnswer(choice) { const feedback = document.getElementById("feedback"); if (choice === currentVerb.type) { feedback.textContent = "✅ 正解！"; feedback.style.color = "green"; } else if (choice === "分からない") { feedback.textContent = `❓ 分からなかったね。正解：${currentVerb.type}`; feedback.style.color = "gray"; wrongAnswers.push(currentVerb); updateWrongList(); } else { feedback.textContent = `❌ 不正解！正解：${currentVerb.type}`; feedback.style.color = "red"; wrongAnswers.push(currentVerb); updateWrongList(); } if (currentVerb.pattern) { document.getElementById("pattern").innerHTML = `🔍 特徴・パターン：${currentVerb.pattern}`; } document.getElementById("translation").textContent = "意味：" + currentVerb.translation; document.getElementById("example").textContent = "例文：" + currentVerb.example; showConjugation(currentVerb.infinitive); } function updateWrongList() { const container = document.getElementById("wrongWords"); container.innerHTML = ""; const unique = [...new Set(wrongAnswers.map(v => v.infinitive))]; unique.forEach(verb => { const div = document.createElement("div"); div.className = "wrong-word"; div.textContent = verb; container.appendChild(div); }); } function nextQuestion() { displayVerb(); } function showConjugation(inf) { const table = document.getElementById("conjugation-table"); const stem = inf.replace(/(se)?$/, "").replace(/arse|erse|irse/, "").replace(/r$/, ""); const irregular = { "ser": ["soy", "eres", "es", "somos", "sois", "son"], "ir": ["voy", "vas", "va", "vamos", "vais", "van"], "tener": ["tengo", "tienes", "tiene", "tenemos", "tenéis", "tienen"], "estar": ["estoy", "estás", "está", "estamos", "estáis", "están"], "hacer": ["hago", "haces", "hace", "hacemos", "hacéis", "hacen"], "decir": ["digo", "dices", "dice", "decimos", "decís", "dicen"], "venir": ["vengo", "vienes", "viene", "venimos", "venís", "vienen"], "ver": ["veo", "ves", "ve", "vemos", "veis", "ven"], "dormir": ["duermo", "duermes", "duerme", "dormimos", "dormís", "duermen"], "preferir": ["prefiero", "prefieres", "prefiere", "preferimos", "preferís", "prefieren"], "sentarse": ["me siento", "te sientas", "se sienta", "nos sentamos", "os sentáis", "se sientan"], "vestirse": ["me visto", "te vistes", "se viste", "nos vestimos", "os vestís", "se visten"] }; const pronouns = ["yo", "tú", "él/ella", "nosotros", "vosotros", "ellos/ellas"]; if (irregular[inf]) { const rows = irregular[inf].map((v, i) => `${pronouns[i]}${v}`).join(""); table.innerHTML = `${rows}
主語現在形
`; } else { let ending; if (inf.endsWith("ar") || inf.endsWith("arse")) { ending = ["o", "as", "a", "amos", "áis", "an"]; } else if (inf.endsWith("er") || inf.endsWith("erse")) { ending = ["o", "es", "e", "emos", "éis", "en"]; } else if (inf.endsWith("ir") || inf.endsWith("irse")) { ending = ["o", "es", "e", "imos", "ís", "en"]; } else { table.innerHTML = "（活用未対応）"; return; } const rows = pronouns.map((p, i) => `${p}${p.includes("yo") ? stem + ending[i] : stem + ending[i]}`).join(""); table.innerHTML = `${rows}
主語現在形
`; } } displayVerb();
